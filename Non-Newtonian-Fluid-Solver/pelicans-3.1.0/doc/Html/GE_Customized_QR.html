SUMMARY :: METHOD
void insert_point(GE_Point const* pt,
double pt_weight )


Precondition
  • !has_been_finalized()
  • pt != 0
  • pt->nb_coordinates() == reference_polyhedron()->dimension()
  • reference_polyhedron()->contains( pt )
  • FORALL( i = 0, i < nb_points(), ++i )
     !pt->is_equal( point( i ) )

Postcondition
  • nb_points() == OLD( nb_points ) + 1
  • point( nb_points() - 1 )->is_equal( pt )
  • point( nb_points() - 1 )->owner() == this
  • weight( nb_points() - 1 ) == pt_weight
