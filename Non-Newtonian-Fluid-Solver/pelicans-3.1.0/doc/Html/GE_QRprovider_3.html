virtual GE_Customized_QR const* create_subset(PEL_Object* a_owner,
GE_Mpolyhedron const* pp,
PEL_Sequence const* poly_list ) const
Create and return a quadrature rule defined on the intersection
of pp and the polyhedra of poly_list for which integration
points are given by self.

Precondition
  • pp != 0
  • pp->dimension() == pp->nb_space_dimensions() || pp->dimension() == pp->nb_space_dimensions() - 1
  • poly_list != 0
  • poly_list->index_limit() == poly_list->count()
  • FORALL( i = 0, i < poly_list->count(), ++i )
     dynamic_cast< GE_Mpolyhedron const* >(poly_list->at( i )) != 0 && static_cast< GE_Mpolyhedron const* >(poly_list->at( i ))->nb_space_dimensions() == pp->nb_space_dimensions() && static_cast< GE_Mpolyhedron const* >(poly_list->at( i ))->dimension() == pp->dimension()
  • FORALL( i = 1, i < poly_list->count(), ++i )
     quadrature_rule( static_cast< GE_Mpolyhedron const* >(poly_list->at( i ))->reference_polyhedron() )->order() == quadrature_rule( static_cast< GE_Mpolyhedron const* >(poly_list->at( 0 ))->reference_polyhedron() )->order()

Postcondition
  • result != 0
  • result->owner() == a_owner
  • result->has_been_finalized()
  • result->reference_polyhedron() == pp->reference_polyhedron()
  • result->order() == quadrature_rule( static_cast< GE_Mpolyhedron const* >(poly_list->at( 0 ))->reference_polyhedron() )->order()
