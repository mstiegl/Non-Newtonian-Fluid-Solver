#ifndef LA_Cholesky_DS_HH
#define LA_Cholesky_DS_HH

#include <LA_Solver.hh>

#include <size_t_vector.hh>

class LA_SeqVector ;
class LA_SymmetricMatrix ;
class LA_SeqMatrix ;
class LA_Vector ;

/*
Linear system solvers using the Cholesky LDLt factorization method.

PUBLISHED
*/

class PEL_EXPORT LA_Cholesky_DS : public LA_Solver
{
   public: //-----------------------------------------------------------

   //-- Instance delivery and initialization
      
      static LA_Cholesky_DS* create( PEL_Object* a_owner,
                                     double const a_pivot_minimal_value ) ;

      virtual LA_Cholesky_DS* create_clone( PEL_Object* a_owner ) const ;
      
   //-- Linear system resolution
      
      // factorized matrix
      LA_SymmetricMatrix const* factorized_LDLt_matrix( void ) const ;
      
      //-- Input - Output

      virtual void print( std::ostream& os, size_t indent_width ) const ;
      
   protected: //--------------------------------------------------------

   private: //----------------------------------------------------------

     ~LA_Cholesky_DS( void ) ;
      LA_Cholesky_DS( LA_Cholesky_DS const& other ) ;
      LA_Cholesky_DS& operator=( LA_Cholesky_DS const& other ) ;

      LA_Cholesky_DS( PEL_Object* a_owner,
                      double const a_pivot_minimal_value ) ;
      LA_Cholesky_DS( PEL_Object* a_owner, LA_Cholesky_DS const* other ) ;
      
   //-- Plug in

      LA_Cholesky_DS( void ) ;
      
      virtual LA_Cholesky_DS* create_replica( 
                                     PEL_Object* a_owner,
                                     PEL_ModuleExplorer const* exp ) const ;

   //-- Linear system resolution
      
      virtual void set_matrix_self( LA_Matrix const* mat,
                                    bool &ok,
                                    bool same_pattern ) ;

      virtual void solve_self( LA_Vector const* b, 
                               LA_Vector* x,
                               size_t &nb_iter,
                               bool &ok ) ;
      
      virtual void unset_matrix_self( void ) ;

   //-- Linear system resolution

      // Do factorization 
      void factorize_LDLt( LA_SeqMatrix const* smat,
                           bool &ok ) ;


   //-- Preconditions, Postconditions, Invariant

      virtual bool invariant( void ) const ;
      
   //-- Class attributes

      static LA_Cholesky_DS const* PROTOTYPE ;
      
   //-- Attributes

      double const PIV_MIN_VAL ;
      
      LA_SymmetricMatrix* const MAT ;
} ;


#endif
