void set_sorted(size_t_vector const& a_repatriated_items_table,
size_t_vector const& a_local_indices_table ) 
Set the items which have to be repatriated
(a_repatriated_items_table is supposed to be sorted).

Precondition
  • FORALL( i = 0, i < a_repatriated_items_table.size(), ++i )
     a_repatriated_items_table( i ) < distribution()->global_number()
  • FORALL( i = 1, i < a_repatriated_items_table.size(), ++i )
     a_repatriated_items_table( i ) > a_repatriated_items_table( i - 1 )
  • a_local_indices_table.size() == a_repatriated_items_table.size()

Postcondition
  • size() == a_repatriated_items_table.size()
  • repatriated_items() == a_repatriated_items_table
  • local_indices() == a_local_indices_table

virtual void get(LA_Vector const* source,
LA_SeqVector* dest ) const
Repatriate distributed items in distributed source to local dest.

Precondition
  • source != 0
  • source->is_synchronized()
  • source->implementation() == implementation()
  • source->row_distribution()->is_compatible( distribution() )
  • dest != 0
  • FORALL( i = 0, i < size(), ++i )
     repatriated_items()( i ) < source->nb_rows()
  • FORALL( i = 0, i < size(), ++i )
     local_indices()( i ) < dest->nb_rows()

Postcondition
  • dest->is_synchronized()
