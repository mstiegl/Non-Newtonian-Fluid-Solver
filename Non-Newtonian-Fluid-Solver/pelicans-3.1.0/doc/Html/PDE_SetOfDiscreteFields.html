void append(PDE_DiscreteField* field,
bool located_on_bounds,
PEL_Vector* elements ) 
Add field.

Precondition
  • field != 0
  • field->owner() == 0
  • elements->owner() == 0
  • elements->index_limit() >= 1
  • elements->count() == elements->index_limit()
  • FORALL( i = 0, i < elements->index_limit(), ++i )
     dynamic_cast< PDE_ReferenceElement const* >(elements->at( i )) != 0
  • !has( field->name() )

Postcondition
  • field->owner() == this
  • elements->owner() == this
  • nb_fields() == OLD( nb_fields ) + 1
  • has( field->name() )
  • item( field->name() ) == field
  • item( index_of( field->name() ) ) == field
  • reference_elements( index_of( field->name() ) ) == elements
  • boundary_located( index_of( field->name() ) ) == located_on_bounds
