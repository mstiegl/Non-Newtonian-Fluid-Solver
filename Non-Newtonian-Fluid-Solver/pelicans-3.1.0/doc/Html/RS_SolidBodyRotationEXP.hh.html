#ifndef RS_SOLID_BODY_ROTATION_EXP_HH
#define RS_SOLID_BODY_ROTATION_EXP_HH

#include <PEL_Expression.hh>

#include <doubleVector.hh> 
#include <doubleArray2D.hh> 

/*
Expressions for a pure advection problem on the domain [-1,1]x[-1,1]

Reference:
   document: Implementation of Solvers for Partial Differential Equations 
             with PELICANS
   chapter:  Reference Solution of Some Systems of 
             Partial Differential Equations
   section:  Linear advection with discontinuous solution II
   
   LeVeque, Randall J., "Finite Volume Methods for Hyperbolic 
      Problems", in Cambridge Texts in Applied Mathematics, 
      Cambridge University Press, 2002, section 20.8.2, pp 461--462
   
PUBLISHED
*/

class PEL_EXPORT RS_SolidBodyRotationEXP : public PEL_Expression
{
   public: //-------------------------------------------------------
      
   //-- Type
      
      virtual PEL_Data::Type data_type( void ) const ;
      
   //-- Value
      
      virtual double to_double( PEL_Context const* ct = 0 ) const ;

      virtual doubleVector const& 
                to_double_vector( PEL_Context const* ct = 0 ) const ;

   protected: //-----------------------------------------------------
      
   private: //-------------------------------------------------------

      RS_SolidBodyRotationEXP( void ) ;
     ~RS_SolidBodyRotationEXP( void ) ;
      RS_SolidBodyRotationEXP( 
                      RS_SolidBodyRotationEXP const& other ) ;
      RS_SolidBodyRotationEXP& operator=( 
                      RS_SolidBodyRotationEXP const& other ) ;

      enum Func { U, B } ;
            
      RS_SolidBodyRotationEXP( PEL_Object* a_owner,
                           std::string const& a_name,
                           PEL_Sequence const* argument_list,
                           Func an_expr ) ;

      PEL_Data const* alternative_result( void ) const ;
      
   //-- Plug in

      RS_SolidBodyRotationEXP( std::string const& a_name, Func an_expr ) ;

      virtual RS_SolidBodyRotationEXP* create_replica( 
                        PEL_Object * a_owner,
                        PEL_Sequence const* argument_list ) const ;

   //-- Characteristics

      virtual std::string const& usage( void ) const ;

      virtual bool valid_arguments( 
                        PEL_Sequence const* some_arguments ) const ;

   //-- Class attributes

      static RS_SolidBodyRotationEXP const* PROTOTYPE_U ;
      static RS_SolidBodyRotationEXP const* PROTOTYPE_B ;

   //-- Attributes

      Func EXPR ;

      mutable doubleVector BETA ;
} ;

#endif
