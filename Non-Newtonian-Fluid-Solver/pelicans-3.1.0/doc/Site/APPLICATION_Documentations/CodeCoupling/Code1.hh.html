#ifndef Code1_HH
#define Code1_HH

#include <PEL_Application.hh>

/*
Application, called "Code1" to be coupled with the application called "Code2"
via the `PEL_CoupledApplications' class.

The global algorithm is the following :
   * "Code2" sends some double scalar T to "Code1"
   * "Code1" computes P = 1.0/T and returns it to "Code2"
   * "Code2" checks that the value of P received from "Code1"
             is equal to 1.0/T and saves it in a file called "save.pel"
*/

class Code1 : public PEL_Application
{

   public: //-----------------------------------------------------------

   //-- Instance delivery and initialization

      static Code1* create( PEL_Object* a_owner,
                            PEL_ModuleExplorer const* exp ) ;
      
   //-- Program core execution

      virtual void run( void ) ;

   protected: //--------------------------------------------------------

   private: //----------------------------------------------------------

     ~Code1( void ) ; 
      Code1( Code1 const& other ) ;
      Code1& operator=( Code1 const& other ) ;

      Code1( PEL_Object* a_owner, PEL_ModuleExplorer const* exp ) ;
      Code1( PEL_Object* a_owner, stringVector& args ) ;
      
   //-- Plug in
      
      Code1( void ) ;
      
      virtual Code1* create_replica( PEL_Object* a_owner,
				     PEL_ModuleExplorer const* exp ) const ;

   //-- Class attributes
      
      static Code1 const* PROTOTYPE ;

   //-- Attributes

      size_t ERROR ;
} ;

#endif
