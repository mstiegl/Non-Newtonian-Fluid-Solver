#ifndef FE_ADAPTATION_INDICATOR_HH
#define FE_ADAPTATION_INDICATOR_HH

#include <PDE_AdaptationIndicator.hh>

class FE_SetOfParameters ;
class FE_TimeIterator ;

/*
FRAMEWORK INSTANTIATION
PUBLISHED
*/

class PEL_EXPORT FE_AdaptationIndicator : public PDE_AdaptationIndicator
{
   public: //-----------------------------------------------------------

   //-- Instance delivery and initialization

      static FE_AdaptationIndicator* make( PEL_Object* a_owner,
                                           PDE_DomainAndFields const* dom,
                                           FE_SetOfParameters const* prms,
                                           PEL_ModuleExplorer const* exp,
                                           size_t a_verbose_level ) ;

   //-- Indicator calculation
      
      void set_time_iterator( FE_TimeIterator const* t_it ) ;
      
      virtual void reset( void ) ;

      virtual void build( void ) ;

   protected: //--------------------------------------------------------

   //-- Plug in

      virtual ~FE_AdaptationIndicator( void ) ;

      FE_AdaptationIndicator( std::string const& a_name ) ;

      FE_AdaptationIndicator( PEL_Object* a_owner,
                              size_t a_verbose_level ) ;

      virtual PDE_AdaptationIndicator* create_replica( 
                                           PEL_Object* a_owner,
                                           PDE_DomainAndFields const* dom,
                                           PEL_ModuleExplorer const* exp,
                                           size_t a_verbose_level ) const ;

      virtual FE_AdaptationIndicator* create_replica( 
                                           PEL_Object* a_owner,
                                           PDE_DomainAndFields const* dom,
                                           FE_SetOfParameters const* prms,
                                           PEL_ModuleExplorer const* exp,
                                           size_t a_verbose_level ) const = 0 ;

   //-- Indicator calculation
         
      virtual void reset_self( void ) = 0 ;

      virtual void build_self( FE_TimeIterator const* t_it ) = 0 ;

   //-- Preconditions, Postconditions, Invariant

      virtual bool create_replica_PRE( PEL_Object* a_owner,
                                       PDE_DomainAndFields const* dom,
                                       FE_SetOfParameters const* prms,
                                       PEL_ModuleExplorer const* exp,
                                       size_t a_verbose_level ) const ;

      virtual bool create_replica_POST( FE_AdaptationIndicator const* result,
                                        PEL_Object* a_owner,
                                        PDE_DomainAndFields const* dom,
                                        FE_SetOfParameters const* prms,
                                        PEL_ModuleExplorer const* exp,
                                        size_t a_verbose_level ) const ;

   private: //----------------------------------------------------------

      FE_AdaptationIndicator( void ) ;
      FE_AdaptationIndicator( FE_AdaptationIndicator const& other ) ;
      FE_AdaptationIndicator& operator=( 
                              FE_AdaptationIndicator const& other ) ;

      static PEL_ObjectRegister* plugins_map( void ) ;
      
   //-- Attributes

      FE_TimeIterator const* T_IT ;
} ;

#endif
