Class LA_DistVector

  • PEL_Object
    • LA_Vector
      • LA_DistVector
   Sequential vectors of double, with a possible block structure in subvectors,
   compatibles with LA_DistMatrix matrices.

   Local items are stored in contiguous vector and non-local
   ones are stored in a sparse structure.

static LA_DistVector* create(PEL_Object* a_owner,
size_t a_nb_rows,
size_t a_nb_local_rows,
LA::DistributionStrategy dist_strat ) 
Create and return an instance.

Precondition
  • dist_strat == LA::FromLocalSize || dist_strat == LA::FromGlobalSize

Postcondition
  • result != 0
  • result->owner() == a_owner
  • FORALL( i = result->row_distribution()->first_local_index(), i < result->row_distribution()->local_index_limit(), ++i )
     result->item( i ) == 0.0
  • result->distribution_strategy() == dist_strat
  • result->distribution_strategy() == LA::FromGlobalSize ==> result->nb_rows() == a_nb_rows
  • result->distribution_strategy() == LA::FromLocalSize ==> result->row_distribution()->local_index_limit() - result->row_distribution()->first_local_index() == a_nb_local_rows
  • result->state() == LA::Sync
  • result->is_resizable()
  • result->is_synchronized()

virtual void set_as_v_product(LA_Vector const* a,
LA_Vector const* b ) 
Reinitialize by assigning to each coefficient the product
of the coefficients of a and b with the same index :
self(i) <- a(i) * b(i) .

Precondition
  • a != 0
  • a->nb_rows() == nb_rows()
  • a->implementation() == implementation()
  • a->is_synchronized()
  • a->row_distribution()->is_compatible( row_distribution() )
  • b != 0
  • b->nb_rows() == nb_rows()
  • b->implementation() == implementation()
  • b->is_synchronized()
  • b->row_distribution()->is_compatible( row_distribution() )

Postcondition
  • is_synchronized()
  • FORMAL( FORALL( i = 0, i < nb_rows(), ++i )
     item( i ) == a->item( i ) * b->item( i ) )
Overridden from LA_Vector SUMMARY :: METHOD
virtual void set_as_reciprocal(LA_Vector const* a,
double smallest_inverted_item = 0.,
double default_value = 1. ) 
Reinitialize by assigning to each coefficient the inverse
of the coefficients of a with the same index.
If such coefficient is such that its magnitude is strictly lower than
smallest_inverted_item then is't value is set to default_value.
Rem : a can be the same than self.

Precondition
  • a != 0
  • a->nb_rows() == nb_rows()
  • a->implementation() == implementation()
  • a->is_synchronized()
  • a->row_distribution()->is_compatible( row_distribution() )
  • smallest_inverted_item >= 0.
  • is_desynchronizable() ==> PEL_Exec::communicator()->same_value_everywhere( smallest_inverted_item )
  • is_desynchronizable() ==> PEL_Exec::communicator()->same_value_everywhere( default_value )

Postcondition
  • is_synchronized()
