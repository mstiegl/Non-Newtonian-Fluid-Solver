#include <PEL.hh>

PDE_2D_Q1isoNonConfB_4nodes const* 
PDE_2D_Q1isoNonConfB_4nodes::REGISTRATOR = new PDE_2D_Q1isoNonConfB_4nodes() ;

//----------------------------------------------------------------------
PDE_2D_Q1isoNonConfB_4nodes:: ~PDE_2D_Q1isoNonConfB_4nodes( void )
//----------------------------------------------------------------------
{
   REGISTRATOR = 0 ;
}

//----------------------------------------------------------------------
PDE_2D_Q1isoNonConfB_4nodes:: PDE_2D_Q1isoNonConfB_4nodes( void )
//----------------------------------------------------------------------
   : PDE_ReferenceElement( "PDE_2D_Q1isoNonConfB_4nodes", 
                           GE_ReferenceSquare::object() )
{
   append_node( GE_Point::create( this, 0.5, 0.0 ) ) ;
   append_node( GE_Point::create( this, 1.0, 0.5 ) ) ;
   append_node( GE_Point::create( this, 0.5, 1.0 ) ) ;
   append_node( GE_Point::create( this, 0.0, 0.5 ) ) ;
}

//----------------------------------------------------------------------
double
PDE_2D_Q1isoNonConfB_4nodes:: N_local( size_t node, 
                                       GE_Point const* pt_ref ) const
//----------------------------------------------------------------------
{
   PEL_LABEL( "PDE_2D_Q1isoNonConfB_4nodes:: N_local" ) ;
   PEL_CHECK_PRE( N_local_PRE( node, pt_ref ) ) ;

   double result = PEL::bad_double() ;

   double x = pt_ref->coordinate( 0 ) ;
   double y = pt_ref->coordinate( 1 ) ;

   switch( node )
   {
      case 0 :
         result = 0.75+x-2.*y-x*x+y*y ;
         break ;
      case 1 :
         result = -0.25+y+x*x-y*y ;
         break ;
      case 2 :
         result = -0.25+x-x*x+y*y ;
         break ;
      case 3 :
         result = 0.75-2.*x+y+x*x-y*y ;
         break ;
   }

   return( result ) ;
}

//----------------------------------------------------------------------
double
PDE_2D_Q1isoNonConfB_4nodes:: dN_local( size_t node, 
                                        size_t a, 
                                        GE_Point const* pt_ref ) const
//----------------------------------------------------------------------
{
   PEL_LABEL( "PDE_2D_Q1isoNonConfB_4nodes:: dN_local" ) ;
   PEL_CHECK_PRE( dN_local_PRE( node, a, pt_ref ) ) ;

   double result = PEL::bad_double() ;

   double x = pt_ref->coordinate( 0 ) ;
   double y = pt_ref->coordinate( 1 ) ;

   switch( node )
   {
      case 0 : // bf = 0.75+x-2y-x²+y² ;
         switch( a )
         {
            case 0 :
               result = 1.-2.*x ;
               break ;
            case 1 :
               result = -2.*(1.0-y) ;
               break ;
         }
         break ;
      case 1 : // bf = -0.25+y+x²-y² ;
         switch( a )
         {
            case 0 :
               result = 2.*x ;
               break ;
            case 1 :
               result = 1.-2.*y ;
               break ;
         }
         break ;
      case 2 : // bf = -0.25+x-x²+y² ;
         switch( a )
         {
            case 0 :
               result = 1.-2.*x ;
               break ;
            case 1 :
               result = 2.*y ;
               break ;
         }
         break ;
      case 3 : // bf = 0.75-2x+y+x²-y² ;
         switch( a )
         {
            case 0 :
               result = -2*(1.-x) ;
               break ;
            case 1 :
               result = 1.-2.*y ;
               break ;
         }
         break ;
   }

   return( result ) ;
}

//----------------------------------------------------------------------
double
PDE_2D_Q1isoNonConfB_4nodes:: d2N_local( size_t node,
				 	 size_t a, 
                                         size_t b,
					 GE_Point const* pt_ref ) const
//----------------------------------------------------------------------
{
   PEL_LABEL( "PDE_2D_Q1isoNonConfB_4nodes:: d2N_local" ) ;
   PEL_CHECK_PRE( d2N_local_PRE( node, a, b, pt_ref ) ) ;

   double result = PEL::bad_double() ;

   switch( node )
   {
      case 0 : // bf = 0.75+x-2y-x²+y² ;
         switch( a )
         {
            case 0 : // dbf = 1.0-2x ;
               switch( b )
               {
                  case 0 :
                     result = -2. ;
                     break ;
                  case 1 :
                     result = 0.0 ;
                     break ;
               }
               break ;
            case 1 : // dbf = -2(1-y)
               switch( b )
               {
                  case 0 :
                     result = 0.0 ;
                     break ;
                  case 1 :
                     result = 2. ;
                     break ;
               }
               break ;
         }
         break ;
      case 1 : // bf = -0.25+y+x²-y² ;
         switch( a )
         {
            case 0 : // dbf = 2.*x ;
               switch( b )
               {
                  case 0 :
                     result = 2. ;
                     break ;
                  case 1 :
                     result = 0. ;
                     break ;
               }
               break ;
            case 1 : // dbf = 1.-2.*y ;
               switch( b )
               {
                  case 0 :
                     result = 0. ;
                     break ;
                  case 1 :
                     result = -2. ;
                     break ;
               }
               break ;
         }
         break ;
      case 2 : // bf = -0.25+x-x²+y² ;
         switch( a )
         {
            case 0 : // dbf = 1.-2.*x ;
               switch( b )
               {
                  case 0 :
                     result = -2. ;
                     break ;
                  case 1 :
                     result = 0. ;
                     break ;
               }
               break ;
            case 1 : // dbf = 2.*y ;
               switch( b )
               {
                  case 0 :
                     result = 0. ;
                     break ;
                  case 1 :
                     result = 2. ;
                     break ;
               }
               break ;
         }
         break ;
      case 3 : // bf = 0.75-2x+y+x²-y² ;
         switch( a )
         {
            case 0 : // dbf = -2*(1.-x) ;
               switch( b )
               {
                  case 0 :
                     result = 2. ;
                     break ;
                  case 1 :
                     result = 0. ;
                     break ;
               }
               break ;
            case 1 : // dbf = 1.-2.*y ;
               switch( b )
               {
                  case 0 :
                     result = 0. ;
                     break ;
                  case 1 :
                     result = -2. ;
                     break ;
               }
               break ;
         }
         break ;
   }

   return( result ) ;
}

//----------------------------------------------------------------------------
bool
PDE_2D_Q1isoNonConfB_4nodes:: reference_polyhedron_POST(
                               GE_ReferencePolyhedron const* result ) const 
//----------------------------------------------------------------------------
{
   PEL_ASSERT( PDE_ReferenceElement::reference_polyhedron_POST( result ) ) ;
   PEL_ASSERT( result == GE_ReferenceSquare::object() ) ;
   return( true ) ;
}
