PDEsolver /
Class PDE_ResultSaver

  • PEL_Object
    • PDE_ResultSaver
Servers that save data in a file for subsequent post-processing.

The resulting file is organized in successive cycles.
Each cycle contains the values of a set of variables.
Variables have a name and may be of different types.

SUMMARY :: METHOD
void save_grid(size_t ndims,
doubleArray2D const& vertices,
intVector const& cell_nb_vertices,
intArray2D const& cell2vertex,
intVector const& cell_nb_faces,
intArray2D const& cell2face,
intVector const& face_nb_vertices,
intArray2D const& face2vertex,
intArray2D const& face2cell,
intVector const& vertex_color,
intVector const& cell_color,
intVector const& face_color )


Precondition
  • has_an_opened_cycle()
  • !field_saving_in_progress()
  • ndims == attached_domain()->nb_space_dimensions()
  • vertices.index_bound( 0 ) == ndims
  • vertices.index_bound( 1 ) != 0
  • cell2vertex.index_bound( 0 ) != 0
  • cell2vertex.index_bound( 1 ) != 0
  • cell2vertex.index_bound( 1 ) == cell_nb_vertices.size()
  • FORALL( ic = 0, ic < cell_nb_vertices.size(), ++ic )
     (cell_nb_vertices( ic ) > 0) && (cell_nb_vertices( ic ) <= cell2vertex.index_bound( 0 ))
  • FORALL( ic = 0, ic < cell_nb_vertices.size(), ++ic )
     FORALL( iv = 0, iv < cell_nb_vertices( ic ), ++iv )
      cell2vertex( iv, ic ) >= 0
  • cell2face.index_bound( 0 ) != 0
  • cell2face.index_bound( 1 ) == cell2vertex.index_bound( 1 )
  • cell2face.index_bound( 1 ) == cell_nb_faces.size()
  • FORALL( ic = 0, ic < cell_nb_faces.size(), ++ic )
     (cell_nb_faces( ic ) > 0) && (cell_nb_faces( ic ) <= cell2face.index_bound( 0 ))
  • FORALL( ic = 0, ic < cell_nb_faces.size(), ++ic )
     FORALL( f = 0, f < cell_nb_faces( ic ), ++f )
      cell2face( f, ic ) >= 0
  • face2vertex.index_bound( 0 ) != 0
  • face2vertex.index_bound( 1 ) != 0
  • face2vertex.index_bound( 1 ) == face_nb_vertices.size()
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     (face_nb_vertices( f ) > 0) && (face_nb_vertices( f ) <= face2vertex.index_bound( 0 ))
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     FORALL( iv = 0, iv < face_nb_vertices( f ), ++iv )
      face2vertex( iv, f ) >= 0
  • face2cell.index_bound( 0 ) == 2
  • face2cell.index_bound( 1 ) == face_nb_vertices.size()
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     face2cell( 0, f ) >= 0
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     (face2cell( 1, f ) == PEL_DataOnMeshingWriter::index_for_trash()) || (face2cell( 1, f ) >= 0)
  • vertex_color.size() == vertices.index_bound( 1 )
  • FORALL( iv = 0, iv < vertex_color.size(), ++iv )
     vertex_color( iv ) >= 0
  • cell_color.size() == cell_nb_vertices.size()
  • FORALL( ic = 0, ic < cell_color.size(), ++ic )
     cell_color( ic ) >= 0
  • face_color.size() == face_nb_vertices.size()
  • FORALL( f = 0, f < face_color.size(), ++f )
     face_color( f ) >= 0

Postcondition
  • grid_is_saved()
  • nb_saved_vertices() == vertices.index_bound( 1 )
  • nb_saved_cells() == cell2vertex.index_bound( 1 )
  • nb_saved_faces() == face2vertex.index_bound( 1 )
SUMMARY :: METHOD
void save_fields(size_t level ) 
For some items of the associated PDE_SetOfDiscreteFields object,
save a variable defined by the values of a reconstruction at
some discrete locations in the geometrical grid (based on the
level-th storage their DOFs).
The considered items, the name of the associated variables, the
discrete locations are given by the PEL_ModuleExplorer object
used when creating self.

Precondition
  • has_an_opened_cycle()
  • grid_is_saved()
  • !field_saving_in_progress()
SUMMARY :: METHOD
enum SavingLocation { AtVertices, AtCellCenters, InvalidLocation }


void prepare_for_field_saving(SavingLocation location,
std::string const& save_name,
size_t nb_components,
doubleArray2D& values,
doubleVector& default_values )


Precondition
  • has_an_opened_cycle()
  • grid_is_saved()
  • !field_saving_in_progress()
  • !save_name.empty()
  • nb_components != 0
  • (location == AtVertices) || (location == AtCellCenters)

Postcondition
  • field_saving_in_progress()
  • saving_location() == location
  • values.index_bound( 0 ) == nb_components
  • location == AtCellCenters ==> values.index_bound( 1 ) == nb_saved_cells()
  • location == AtVertices ==> values.index_bound( 1 ) == nb_saved_vertices()
  • FORALL( ic = 0, ic < values.index_bound( 0 ), ++ic )
     FORALL( im = 0, im < values.index_bound( 1 ), ++im )
      values( ic, im ) == PDE_ResultSaver::undefined_value()
  • default_values.size() == nb_components
  • FORALL( ic = 0, ic < default_values.size(), ++ic )
     default_values( ic ) == PDE_ResultSaver::undefined_value()

SUMMARY :: METHOD
static void check_value_consistency_at_vertex(std::string const& banner,
GE_Point const* vertex,
double old_val,
double new_val,
double dbl_eps,
double dbl_min ) 
      Check that the two values old_val and new_val at the same
      vertex vertex (but from two different cells sharing that vertex) 
      are identical.
      Remarks :
         - no check is performed if old_val = undefined_value ;
         - the equality of old_val and new_val is checked calling
           PEL::double_equality with dbl_eps and dbl_min as 
            last two parameters.

Precondition
  • vertex != 0
  • dbl_eps > 0.
  • dbl_min > 0.
