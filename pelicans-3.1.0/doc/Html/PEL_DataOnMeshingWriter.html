   1. Derive a concrete subclass.
   2. Create a static class pointer which defines the model of the concrete
      class (prototype). It is built calling the prototype constructor :
          PEL_DataOnMeshingWriter
   3. Implement the function :
          create_replica
      which create a new instance of the concrete class, calling the constructor :
        PEL_DataOnMeshingWriter
   4. Implement a destructor
   5. Implement the virtual function :
          write_cycle

std::string const& halo_color_name,
intArray2D const& macro_colors,
intVector const& vertex_color,
intVector const& cell_color,
intVector const& face_color )


Precondition
  • !meshing_name.empty()
  • vertices.index_bound( 0 ) == nb_sp_dims
  • vertices.index_bound( 1 ) != 0
  • cell2vertex.index_bound( 0 ) != 0
  • cell2vertex.index_bound( 1 ) != 0
  • cell2vertex.index_bound( 1 ) == cell_nb_vertices.size()
  • FORALL( ic = 0, ic < cell_nb_vertices.size(), ++ic )
     (cell_nb_vertices( ic ) > 0) && (cell_nb_vertices( ic ) <= cell2vertex.index_bound( 0 ))
  • FORALL( ic = 0, ic < cell_nb_vertices.size(), ++ic )
     FORALL( iv = 0, iv < cell_nb_vertices( ic ), ++iv )
      cell2vertex( iv, ic ) >= 0
  • cell2face.index_bound( 0 ) != 0
  • cell2face.index_bound( 1 ) == cell2vertex.index_bound( 1 )
  • cell2face.index_bound( 1 ) == cell_nb_faces.size()
  • FORALL( ic = 0, ic < cell_nb_faces.size(), ++ic )
     (cell_nb_faces( ic ) > 0) && (cell_nb_faces( ic ) <= cell2face.index_bound( 0 ))
  • FORALL( ic = 0, ic < cell_nb_faces.size(), ++ic )
     FORALL( f = 0, f < cell_nb_faces( ic ), ++f )
      cell2face( f, ic ) >= 0
  • face2vertex.index_bound( 0 ) != 0
  • face2vertex.index_bound( 1 ) != 0
  • face2vertex.index_bound( 1 ) == face_nb_vertices.size()
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     (face_nb_vertices( f ) > 0) && (face_nb_vertices( f ) <= face2vertex.index_bound( 0 ))
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     FORALL( iv = 0, iv < face_nb_vertices( f ), ++iv )
      face2vertex( iv, f ) >= 0
  • face2cell.index_bound( 0 ) == 2
  • face2cell.index_bound( 1 ) == face_nb_vertices.size()
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     face2cell( 0, f ) >= 0
  • FORALL( f = 0, f < face_nb_vertices.size(), ++f )
     (face2cell( 1, f ) == index_for_trash()) || (face2cell( 1, f ) >= 0)
  • color_table.size() != 0
  • EXISTS( i = 0, i < color_table.size(), ++i )
     halo_color_name == color_table( i )
  • vertex_color.size() == vertices.index_bound( 1 )
  • FORALL( iv = 0, iv < vertex_color.size(), ++iv )
     (vertex_color( iv ) >= 0) && (vertex_color( iv ) < color_table.size())
  • cell_color.size() == cell_nb_vertices.size()
  • FORALL( ic = 0, ic < cell_color.size(), ++ic )
     (cell_color( ic ) >= 0) && (cell_color( ic ) < color_table.size())
  • face_color.size() == face_nb_vertices.size()
  • FORALL( f = 0, f < face_color.size(), ++f )
     (face_color( f ) >= 0) && (face_color( f ) < color_table.size())

Postcondition
  • result != 0
  • result->owner() == a_owner
  • result->name() == "meshing"
