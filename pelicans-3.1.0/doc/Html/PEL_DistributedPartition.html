void set(PEL_DistributedPartition const* other )


Precondition
  • other != 0

Postcondition
  • first_local_index() == other->first_local_index()
  • local_index_limit() == other->local_index_limit()
  • global_number() == other->global_number()
  • local_number() == other->local_number()
  • partitioning() == other->partitioning()
  • start_of_partition() == other->start_of_partition()
SUMMARY :: METHOD
void set_local_number(size_t a_local_number )


Postcondition
  • local_number() == a_local_number
  • global_number() == communicator()->sum( a_local_number )
  • partitioning()( communicator()->rank() ) == a_local_number
  • partitioning().sum() == global_number()
SUMMARY :: METHOD
void distribute_global_number(size_t a_global_number )


Postcondition
  • global_number() == a_global_number
  • partitioning().sum() == global_number()
  • local_number() == a_global_number / communicator()->nb_ranks() || local_number() == a_global_number / communicator()->nb_ranks() + 1
SUMMARY :: METHOD
void set_global_number(size_t a_global_number )


Postcondition
  • global_number() == a_global_number
  • partitioning().sum() == global_number()
  • partitioning()( communicator()->rank() ) == a_global_number
  • local_number() == a_global_number
  • first_local_index() == 0
  • local_index_limit() == a_global_number
