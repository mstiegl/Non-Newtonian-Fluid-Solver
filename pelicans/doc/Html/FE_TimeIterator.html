static FE_TimeIterator* create(PEL_Object* a_owner,
PEL_ModuleExplorer const* exp )


Precondition
  • exp != 0

Postcondition
  • result != 0
  • result->owner() == a_owner
  • result->initial_time() == exp->double_data( "time_initial" )
  • result->final_time() == exp->double_data( "time_end" )
  • !result->time_step_is_fixed() ==> result->time_step() == exp->double_data( "time_step" )
  • result->initial_iteration_number() == 1
  • exp->has_entry( "storage_depth" ) ==> result->storage_depth() == exp->int_data( "storage_depth" )
  • !exp->has_entry( "storage_depth" ) ==> result->storage_depth() == 5
  • result->time() == result->initial_time()
  • !result->is_started()
  • result->time() == result->initial_time()

SUMMARY :: METHOD
void go_next_time(void ) 
Advance one time step.

Precondition
  • is_started()

Postcondition
  • is_finished() ==> iteration_number() == OLD( iteration_number )
  • !is_finished() ==> iteration_number() == OLD( iteration_number ) + 1
  • is_finished() ==> time() == OLD( time )
  • is_finished() ==> time_step() == OLD( time_step )
  • !is_finished() ==> FORMAL( time() == OLD( time ) + time_step() )
  • !is_finished() ==> !just_went_back()
  • !is_finished() ==> just_went_back_and_forth() == OLD( just_went_back )
