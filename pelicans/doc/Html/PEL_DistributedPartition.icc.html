//----------------------------------------------------------------------
inline
size_t
PEL_DistributedPartition:: rank_of( size_t i ) const
//----------------------------------------------------------------------
{
   PEL_LABEL( "PEL_DistributedPartition:: rank_of" ) ;
   PEL_CHECK_PRE( i < global_number() ) ;

   size_t result = 0 ;
   while( result<SIZE-1 && (size_t)START( result+1 ) <= i ) result++ ;
   
   PEL_CHECK_POST( result < SIZE ) ;
   PEL_CHECK_POST( ( (size_t)START(result) <= i ) && ( (size_t)START(result)+(size_t)PARTITION(result)>i ) ) ;
   
   return( result ) ;
}
